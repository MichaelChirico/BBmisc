  rec.lapply = function(xs, fun, depth=Inf) {
    if (!is.list(xs) || is.data.frame(xs) || depth==0) {
      return(fun(xs))
    }
    lapply(xs, function(x) rec.lapply(x, fun, depth-1))
  }
